async function loadListings(){ const keyword = document.getElementById("searchInput").value.toLowerCase(); const filter = document.getElementById("filterType").value; let query = client.from("listings").select("*"); if(filter){ query = query.eq("type", filter); } const {data} = await query; const c = document.getElementById("listings"); c.innerHTML = ""; data .filter(item => item.title?.toLowerCase().includes(keyword) || item.category?.toLowerCase().includes(keyword) || item.location?.toLowerCase().includes(keyword) ) .forEach(i=>{ const d=document.createElement("div"); d.className="card"; d.innerHTML = `

${i.title}

${i.price}

${i.location}

`; c.appendChild(d); }); }
